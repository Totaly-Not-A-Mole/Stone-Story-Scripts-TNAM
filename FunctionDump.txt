//mindstone dodge
func dodge(S, T)//state, Animation time
  ?foe.state = S & foe.time = T
        ?item.GetCooldown("mind") <= 0
          equipL mind stone
          equipR compound shield
//blade of the fallen god ability
func godBlade(C)//Count
  ?foe.count >= C
    ?item.getcooldown("blade") <= 0
      equip blade of the fallen god
      activate right
//bardiche ability
func Bardiche()
  ?foe.distance <= 10 //change this line as needed
  ^ & item.CanActivate()
  ^ & item.GetCooldown("bardiche") <= 0
    equip bardiche
    activate R
  ?item.right = "bardiche" //keeps it equipped during the cast
  ^ & item.right.state = 2
    equip bardiche
    activate R
//heavy hammer ability
func Hammer()
  ?foe.distance < 12 //change this line as needed
  ^ & item.CanActivate()
  ^ & item.GetCooldown("heavy_hammer") <= 0
    equip heavy hammer
    activate R
  ?item.right = "heavy_hammer" //keeps it equipped during the cast
  ^ & item.right.state = 2
    equip heavy hammer
    activate R
//dashing shield ability
func dash()
  ?item.GetCooldown("dash") <= 0
    equip dash
    activate right
//summon cinderwisp
func summonCinder(sCnt)//summon count
  ?summon.count = sCnt &
  ^item.CanActivate() |
  ^item.right = "fire talisman" &
  ^item.right.state = 2
    equipR fire talisman
    activate right
//cinderwisp ability
func cinder(C)//debuff count
  var deBuff
  deBuff = "Ignition:" + C
  ?item.getcooldown("cinderwisp") <= 0
  ^ & foe.debuffs.string = deBuff
    activate cinderwisp
//error function    
func error(msg)
  >`20,2,ERROR: @msg@
//AAC
func AAC()
  ?item.left.state = 3
    equipL *0 wand //trash
    equipL @item.left@
  ?item.right.state = 3
    equipR *0 shield //trash
    equip @item.right@
//HitPoint Percentage
func HPP()
  return (hp * 100) / maxhp

//Armor Percentage
func ARP()
  var maxarmorMax = storage.Get("maxarmorMax", 1)
  ?maxarmor > maxarmorMax
    maxarmorMax = maxarmor
    storage.Set("maxarmorMax", maxarmorMax)
  ?maxarmor > 0
    return (armor * 100) / maxarmor 
  :
    return (armor * 100) / maxarmorMax
